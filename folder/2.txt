def two_sum(arr, target):
    x={}
    li=[]
    for i in range(len(arr)):
        need=target-arr[i]
        if need in x:
            li.append(i)
            li.append(x.get(need))
            li.sort()
            return li
        else:
            x[arr[i]]=i
    return li
